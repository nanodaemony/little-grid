# 时间线功能实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**目标：** 实现一个完整的时间线功能，允许用户创建和管理多个时间线集合，每个集合包含按时间顺序排列的事件节点，支持自动排序和手动排序两种模式。

**架构：** 后端使用 Spring Boot + JPA + MySQL，前端使用 Flutter + Provider + http，通过 REST API 通信，用户数据通过 userId 隔离。

**技术栈：** Spring Boot 3.2.5, JPA, MySQL, Flutter 3.0+, Provider, http, Knife4j

```java
package me.zhengjie.modules.tools.service;

import lombok.RequiredArgsConstructor;
import me.zhengjie.modules.tools.domain.TimelineCollection;
import me.zhengjie.modules.tools.domain.TimelineNode;
import me.zhengjie.modules.tools.enums.SortMode;
import me.zhengjie.modules.tools.repository.TimelineCollectionRepository;
import me.zhengjie.modules.tools.repository.TimelineNodeRepository;
import me.zhengjie.modules.tools.service.dto.*;
import me.zhengjie.modules.tools.utils.DtoConverter;
import me.zhengjie.exception.BadRequestException;
import me.zhengjie.exception.EntityNotFoundException;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class TimelineService {

    private final TimelineCollectionRepository collectionRepository;
    private final TimelineNodeRepository nodeRepository;

    /**
     * 创建时间线集合
     */
    public TimelineCollectionDTO createCollection(Long userId, CreateCollectionDTO dto) {
        TimelineCollection collection = new TimelineCollection();
        collection.setUserId(userId);
        collection.setTitle(dto.getTitle());
        collection.setDescription(dto.getDescription());
        collection.setSortMode(SortMode.valueOf(dto.getSortMode()));

        TimelineCollection saved = collectionRepository.save(collection);
        return DtoConverter.toDTO(saved);
    }

    /**
     * 获取用户所有时间线集合
     */
    public List<TimelineCollectionDTO> getCollections(Long userId) {
        List<TimelineCollection> collections = collectionRepository.findByUserIdOrderByUpdatedAtDesc(userId);
        return collections.stream()
                .map(DtoConverter::toDTO)
                .collect(Collectors.toList());
    }

    /**
     * 获取单个集合详情
     */
    public TimelineCollectionDTO getCollection(Long collectionId, Long userId) {
        TimelineCollection collection = collectionRepository.findByIdAndUserId(collectionId, userId);
        if (collection == null) {
            throw new EntityNotFoundException(TimelineCollection.class, collectionId);
        }
        return DtoConverter.toDTO(collection);
    }

    /**
     * 更新集合信息
     */
    @Transactional
    public TimelineCollectionDTO updateCollection(Long collectionId, Long userId, UpdateCollectionDTO dto) {
        TimelineCollection collection = collectionRepository.findByIdAndUserId(collectionId, userId);
        if (collection == null) {
            throw new EntityNotFoundException(TimelineCollection.class, collectionId);
        }

        boolean needAutoSort = false;
        SortMode oldSortMode = collection.getSortMode();

        if (dto.getTitle() != null) {
            collection.setTitle(dto.getTitle());
        }
        if (dto.getDescription() != null) {
            collection.setDescription(dto.getDescription());
        }
        if (dto.getSortMode() != null) {
            SortMode newSortMode = SortMode.valueOf(dto.getSortMode());
            collection.setSortMode(newSortMode);
            // 如果从手动切换到自动，需要重新按事件时间排序
            if (oldSortMode == SortMode.MANUAL && newSortMode == SortMode.AUTO) {
                needAutoSort = true;
            }
        }

        TimelineCollection saved = collectionRepository.save(collection);

        // 如果需要自动排序，更新所有节点的sortOrder
        if (needAutoSort) {
            List<TimelineNode> nodes = nodeRepository.findByCollectionIdOrderByEventTimeAsc(collectionId);
            for (int i = 0; i < nodes.size(); i++) {
                nodes.get(i).setSortOrder(i);
            }
            nodeRepository.saveAll(nodes);
        }

        return DtoConverter.toDTO(saved);
    }

    /**
     * 删除集合及其所有节点
     */
    @Transactional
    public void deleteCollection(Long collectionId, Long userId) {
        TimelineCollection collection = collectionRepository.findByIdAndUserId(collectionId, userId);
        if (collection == null) {
            throw new EntityNotFoundException(TimelineCollection.class, collectionId);
        }

        // 先删除所有节点
        nodeRepository.deleteByCollectionIdAndUserId(collectionId, userId);
        // 再删除集合
        collectionRepository.delete(collection);
    }

    /**
     * 添加时间节点
     */
    public TimelineNodeDTO createNode(Long collectionId, Long userId, CreateNodeDTO dto) {
        TimelineCollection collection = collectionRepository.findByIdAndUserId(collectionId, userId);
        if (collection == null) {
            throw new EntityNotFoundException(TimelineCollection.class, collectionId);
        }

        TimelineNode node = new TimelineNode();
        node.setCollectionId(collectionId);
        node.setUserId(userId);
        node.setTitle(dto.getTitle());
        node.setContent(dto.getContent());
        node.setEventTime(dto.getEventTime());

        // 如果是自动排序模式，设置sortOrder为当前最大值+1
        if (collection.getSortMode() == SortMode.AUTO) {
            List<TimelineNode> nodes = nodeRepository.findByCollectionIdOrderByEventTimeAsc(collectionId);
            int maxSortOrder = nodes.isEmpty() ? 0 : nodes.get(nodes.size() - 1).getSortOrder() + 1;
            node.setSortOrder(maxSortOrder);
        } else {
            // 手动排序模式，设置为列表末尾
            List<TimelineNode> nodes = nodeRepository.findByCollectionIdOrderBySortOrderAsc(collectionId);
            int maxSortOrder = nodes.isEmpty() ? 0 : nodes.get(nodes.size() - 1).getSortOrder() + 1;
            node.setSortOrder(maxSortOrder);
        }

        TimelineNode saved = nodeRepository.save(node);
        return DtoConverter.toDTO(saved);
    }

    /**
     * 获取集合的所有节点
     */
    public List<TimelineNodeDTO> getNodes(Long collectionId, Long userId) {
        TimelineCollection collection = collectionRepository.findByIdAndUserId(collectionId, userId);
        if (collection == null) {
            throw new EntityNotFoundException(TimelineCollection.class, collectionId);
        }

        List<TimelineNode> nodes;
        if (collection.getSortMode() == SortMode.AUTO) {
            nodes = nodeRepository.findByCollectionIdOrderByEventTimeAsc(collectionId);
        } else {
            nodes = nodeRepository.findByCollectionIdOrderBySortOrderAsc(collectionId);
        }

        return nodes.stream()
                .map(DtoConverter::toDTO)
                .collect(Collectors.toList());
    }

    /**
     * 获取单个节点详情
     */
    public TimelineNodeDTO getNode(Long nodeId, Long userId) {
        TimelineNode node = nodeRepository.findByIdAndUserId(nodeId, userId);
        if (node == null) {
            throw new EntityNotFoundException(TimelineNode.class, nodeId);
        }
        return DtoConverter.toDTO(node);
    }

    /**
     * 更新节点内容
     */
    public TimelineNodeDTO updateNode(Long nodeId, Long userId, UpdateNodeDTO dto) {
        TimelineNode node = nodeRepository.findByIdAndUserId(nodeId, userId);
        if (node == null) {
            throw new EntityNotFoundException(TimelineNode.class, nodeId);
        }

        if (dto.getTitle() != null) {
            node.setTitle(dto.getTitle());
        }
        if (dto.getContent() != null) {
            node.setContent(dto.getContent());
        }
        if (dto.getEventTime() != null) {
            node.setEventTime(dto.getEventTime());
        }
        if (dto.getSortOrder() != null) {
            node.setSortOrder(dto.getSortOrder());
        }

        TimelineNode saved = nodeRepository.save(node);
        return DtoConverter.toDTO(saved);
    }

    /**
     * 删除节点
     */
    public void deleteNode(Long nodeId, Long userId) {
        TimelineNode node = nodeRepository.findByIdAndUserId(nodeId, userId);
        if (node == null) {
            throw new EntityNotFoundException(TimelineNode.class, nodeId);
        }
        nodeRepository.delete(node);
    }

    /**
     * 调整节点顺序（手动排序）
     */
    @Transactional
    public void reorderNodes(Long collectionId, Long userId, List<Long> nodeIds) {
        TimelineCollection collection = collectionRepository.findByIdAndUserId(collectionId, userId);
        if (collection == null) {
            throw new EntityNotFoundException(TimelineCollection.class, collectionId);
        }

        if (collection.getSortMode() != SortMode.MANUAL) {
            throw new BadRequestException("只能在手动排序模式下调整顺序");
        }

        for (int i = 0; i < nodeIds.size(); i++) {
            Long nodeId = nodeIds.get(i);
            TimelineNode node = nodeRepository.findByIdAndUserId(nodeId, userId);
            if (node != null && node.getCollectionId().equals(collectionId)) {
                node.setSortOrder(i);
            }
        }
    }
}
```

```dart
  bool get hasCurrentCollection => _currentCollection != null;
  bool get hasNodes => _nodes.isNotEmpty;

  // ============ 集合操作 ============

  Future<void> loadCollections() async {
    _isLoading = true;
    _errorMessage = null;
    notifyListeners();

    try {
      _collections = await TimelineService.getCollections();

      // 如果之前有选中的集合，尝试恢复选中状态
      if (_currentCollection != null) {
        final found = _collections.firstWhere(
          (c) => c.id == _currentCollection!.id,
          orElse: () => _collections.firstOrNull,
        );
        if (found != null) {
          _currentCollection = found;
          await loadNodes(found.id);
        } else {
          _currentCollection = null;
          _nodes = [];
        }
      }

      _isLoading = false;
      notifyListeners();
    } catch (e) {
      _isLoading = false;
      _errorMessage = e.toString();
      notifyListeners();
    }
  }

  Future<void> createCollection({
    required String title,
    String? description,
    SortMode sortMode = SortMode.auto,
  }) async {
    _errorMessage = null;
    notifyListeners();

    try {
      final collection = await TimelineService.createCollection(
        title: title,
        description: description,
        sortMode: sortMode,
      );

      _collections.insert(0, collection);
      _currentCollection = collection;
      await loadNodes(collection.id);

      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  Future<void> updateCollection(
    int id, {
    String? title,
    String? description,
    SortMode? sortMode,
  }) async {
    _errorMessage = null;
    notifyListeners();

    try {
      final updated = await TimelineService.updateCollection(
        id,
        title: title,
        description: description,
        sortMode: sortMode,
      );

      final index = _collections.indexWhere((c) => c.id == id);
      if (index != -1) {
        _collections[index] = updated;
        if (_currentCollection?.id == id) {
          _currentCollection = updated;
          // 如果排序模式改变了，重新加载节点
          if (sortMode != null) {
            await loadNodes(id);
          }
        }
      }

      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  Future<void> deleteCollection(int id) async {
    _errorMessage = null;
    notifyListeners();

    try {
      await TimelineService.deleteCollection(id);

      _collections.removeWhere((c) => c.id == id);
      if (_currentCollection?.id == id) {
        _currentCollection = null;
        _nodes = [];
      }

      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  void selectCollection(TimelineCollection collection) async {
    if (_currentCollection?.id == collection.id) return;

    _currentCollection = collection;
    notifyListeners();

    await loadNodes(collection.id);
  }

  // ============ 节点操作 ============

  Future<void> loadNodes(int collectionId) async {
    _isLoading = true;
    _errorMessage = null;
    notifyListeners();

    try {
      _nodes = await TimelineService.getNodes(collectionId);
      _isLoading = false;
      notifyListeners();
    } catch (e) {
      _isLoading = false;
      _errorMessage = e.toString();
      notifyListeners();
    }
  }

  Future<void> createNode({
    required String title,
    String? content,
    DateTime? eventTime,
  }) async {
    if (_currentCollection == null) return;

    _errorMessage = null;
    notifyListeners();

    try {
      final node = await TimelineService.createNode(
        _currentCollection!.id,
        title: title,
        content: content,
        eventTime: eventTime,
      );

      // 根据排序模式添加到正确位置
      if (_currentCollection!.sortMode == SortMode.auto) {
        _nodes.add(node);
        _nodes.sort((a, b) {
          if (a.eventTime == null && b.eventTime == null) return 0;
          if (a.eventTime == null) return 1;
          if (b.eventTime == null) return -1;
          return a.eventTime!.compareTo(b.eventTime!);
        });
      } else {
        _nodes.add(node);
      }

      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  Future<void> updateNode(
    int id, {
    String? title,
    String? content,
    DateTime? eventTime,
  }) async {
    _errorMessage = null;
    notifyListeners();

    try {
      final updated = await TimelineService.updateNode(
        id,
        title: title,
        content: content,
        eventTime: eventTime,
      );

      final index = _nodes.indexWhere((n) => n.id == id);
      if (index != -1) {
        _nodes[index] = updated;
        // 如果是自动排序模式且修改了时间，重新排序
        if (_currentCollection?.sortMode == SortMode.auto && eventTime != null) {
          _nodes.sort((a, b) {
            if (a.eventTime == null && b.eventTime == null) return 0;
            if (a.eventTime == null) return 1;
            if (b.eventTime == null) return -1;
            return a.eventTime!.compareTo(b.eventTime!);
          });
        }
      }

      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  Future<void> deleteNode(int id) async {
    _errorMessage = null;
    notifyListeners();

    try {
      await TimelineService.deleteNode(id);
      _nodes.removeWhere((n) => n.id == id);
      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  Future<void> reorderNodes(List<int> nodeIds) async {
    if (_currentCollection == null) return;

    _errorMessage = null;
    notifyListeners();

    try {
      await TimelineService.reorderNodes(_currentCollection!.id, nodeIds);

      // 根据新的顺序重新排列节点
      final Map<int, TimelineNode> nodeMap = {for (var n in _nodes) n.id: n};
      _nodes = nodeIds.map((id) => nodeMap[id]!).toList();

      notifyListeners();
    } catch (e) {
      _errorMessage = e.toString();
      notifyListeners();
      rethrow;
    }
  }

  void clearError() {
    _errorMessage = null;
    notifyListeners();
  }
}
```

```dart
  }

  Future<void> _submit() async {
    if (!_formKey.currentState!.validate()) return;

    final provider = context.read<TimelineProvider>();
    try {
      if (widget.existingNode != null) {
        await provider.updateNode(
          widget.existingNode!.id,
          title: _titleController.text.trim(),
          content: _contentController.text.trim().isEmpty
              ? null
              : _contentController.text.trim(),
          eventTime: _selectedDateTime,
        );
      } else {
        await provider.createNode(
          title: _titleController.text.trim(),
          content: _contentController.text.trim().isEmpty
              ? null
              : _contentController.text.trim(),
          eventTime: _selectedDateTime,
        );
      }
      if (mounted) Navigator.pop(context);
    } catch (e) {
      _showError(e.toString());
    }
  }

  void _showError(String message) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text(message)),
    );
  }

  Future<void> _selectDateTime() async {
    final now = DateTime.now();
    final picked = await showDatePicker(
      context: context,
      initialDate: _selectedDateTime ?? now,
      firstDate: DateTime(2000),
      lastDate: DateTime(2100),
    );

    if (picked != null && mounted) {
      final timePicked = await showTimePicker(
        context: context,
        initialTime: _selectedDateTime != null
            ? TimeOfDay.fromDateTime(_selectedDateTime!)
            : TimeOfDay.now(),
      );

      if (timePicked != null) {
        setState(() {
          _selectedDateTime = DateTime(
            picked!.year,
            picked!.month,
            picked!.day,
            timePicked!.hour,
            timePicked!.minute,
          );
        });
      }
    }
  }

  String _formatDateTime(DateTime? dateTime) {
    if (dateTime == null) return '未设置';
    return '${dateTime!.year}-${dateTime!.month.toString().padLeft(2, '0')}-${dateTime!.day.toString().padLeft(2, '0')} ${dateTime!.hour.toString().padLeft(2, '0')}:${dateTime!.minute.toString().padLeft(2, '0')}';
  }

  @override
  Widget build(BuildContext context) {
    final isEdit = widget.existingNode != null;

    return AlertDialog(
      title: Text(is(isEdit ? '编辑时间节点' : '新增时间节点')),
      content: SingleChildScrollView(
        child: Form(
          key: _formKey,
          child: Column(
            mainAxisSize: MainAxisSize.min,
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              const SizedBox(height: 16),
              const Text('事件时间（可选）', style: TextStyle(fontWeight: FontWeight.bold)),
              const SizedBox(height: 8),
              InkWell(
                onTap: _selectDateTime,
                child: Container(
                  padding: const EdgeInsets.symmetric(vertical: 12, horizontal: 16),
                  decoration: BoxDecoration(
                    border: Border.all(color: Colors.grey[300]!),
                    borderRadius: BorderRadius.circular(4),
                  ),
                  child: Row(
                    children: [
                      const Icon(Icons.calendar_today),
                      const SizedBox(width: 8),
                      Expanded(
                        child: Text(_formatDateTime(_selectedDateTime)),
                      ),
                      const Icon(Icons.arrow_drop_down),
                    ],
                  ),
                ),
              ),
              const SizedBox(height: 16),
              const Text('标题', style: TextStyle(fontWeight: FontWeight.bold)),
              const SizedBox(height: 8),
              TextFormField(
                controller: _titleController,
                decoration: const InputDecoration(
                  hintText: '请输入标题',
                  border: OutlineInputBorder(),
                ),
                maxLength: 100,
                validator: (value) {
                  if (value == null || value.trim().isEmpty) {
                    return '请输入标题';
                  }
                  return null;
                },
              ),
              const SizedBox(height: 16),
              const Text('内容（可选）', style: TextStyle(fontWeight: FontWeight.bold)),
              const SizedBox(height: 8),
              TextFormField(
                controller: _contentController,
                decoration: const InputDecoration(
                  hintText: '请输入内容',
                  border: OutlineInputBorder(),
                ),
                maxLines: 5,
                maxLength: 5000,
              ),
            ],
          ),
        ),
      ),
      actions: [
        TextButton(
          onPressed: () => Navigator.pop(context),
          child: const Text('取消'),
        ),
        ElevatedButton(
          onPressed: _submit,
          child: Text(isEdit ? '保存' : '创建'),
        ),
      ],
    );
  }
}
```
